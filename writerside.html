Quick Start

Dockerfile

FROM jetbrains/writerside-builder:243.22562 CMD export DISPLAY=:99 && \ Xvfb :99 & \ /opt/builder/bin/idea.sh helpbuilderinspect --source-dir /opt/sources --product $INSTANCE --runner other --output-dir /opt/sources/output

docker build

docker build -f Dockerfile -t registry.cn-shanghai.aliyuncs.com/iproute/wrs-builder:243.22562 .

docker run

instance="Writerside/blog" docker run --rm \ -v "$build_dir:/opt/sources" \ -e INSTANCE="$instance" \ "registry.cn-shanghai.aliyuncs.com/iproute/wrs-builder:243.21565"
